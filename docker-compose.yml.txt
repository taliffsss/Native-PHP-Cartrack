version: "3.7"

services:

  nginx:
    build:
      context: .
      dockerfile: nginx/Dockerfile
    networks: 
      - user-svc
    container_name: nginx-slim
    restart: on-failure
    image: nginx-slim
    depends_on:
     - php
    ports:
      - "8081:80"
    networks: 
      - cartrack
    environment:
      PHP_HOST: php
      PHP_PORT: 9002
    volumes:
      - ./App:/var/www

  php:
    build:
      context: .
      dockerfile: php-fpm/Dockerfile
    networks: 
      - cartrack
    container_name: php-slim
    ports:
      - "9002:9000"
    restart: on-failure
    image: php-slim
    environment:
      REDIS_HOST: redis
      REDIS_PORT: 63711
      MYSQL_HOST: mysql
    depends_on:
      - redis
      - mysql
    volumes:
      - ./App:/var/www

  mysql:
    image: mysql:5.7.29
    networks: 
      - cartrack
    container_name: mysql-slim
    restart: on-failure
    tty: true
    ports:
      - "3308:3306"
    environment:
      MYSQL_DATABASE: cartrack-db
      MYSQL_USER: docker_local
      MYSQL_PASSWORD: secret
      MYSQL_ROOT_PASSWORD: secret
      SERVICE_TAGS: dev
      SERVICE_NAME: mysql
    volumes:
      - ./mysql-data:/var/lib/mysql

  redis:
    image: redis:6-alpine
    networks: 
      - cartrack
    container_name: redis-slim
    ports:
      - 63711:6379
    restart: "on-failure"
    environment:
      REDIS_HOST: redis
      REDIS_PORT: 63711

networks:
  cartrack:
